  <script>
    const outputDiv = document.getElementById("output");

    // Function to fetch todos from API
    function fetchTodo() {
      return fetch('https://jsonplaceholder.typicode.com/todos')
        .then(response => {
          if (!response.ok) {
            throw new Error('Network response was not ok ' + response.statusText);
          }
          return response.json();
        });
    }

    // Function to load and display todos
    function loadTodos() {
      outputDiv.innerHTML = "<p>Loading todos... ‚è≥</p>";
      
      fetchTodo()
        .then(data => {
          outputDiv.innerHTML = "<h3>‚úÖ Todo items fetched successfully:</h3>";
          // show only first 10 items for simplicity
          data.slice(0, 10).forEach(todo => {
            const todoDiv = document.createElement("div");
            todoDiv.className = "todo-item";
            todoDiv.innerHTML = `<strong>${todo.title}</strong> ‚Äî Status: ${todo.completed ? "‚úÖ Completed" : "‚ùå Pending"}`;
            outputDiv.appendChild(todoDiv);
          });
        })
        .catch(error => {
          outputDiv.innerHTML = `<p style="color:red;">‚ö†Ô∏è There was a problem with the fetch operation: ${error.message}</p>`;
        });
    }
  </script>
